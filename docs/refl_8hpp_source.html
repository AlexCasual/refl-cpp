92  /**
93  * Represents a compile-time string. Used in refl-cpp
94  * for representing names of reflected types and members.
95  * Supports constexpr concatenation and substring,
96  * and is explicitly-convertible to const char* and std::string.
97  * REFL_MAKE_CONST_STRING can be used to create an instance from a literal string.
98  *
99  * @param <N> The length of the string excluding the terminating '\0' character.
100  */
101  template <size_t N>
102  struct const_string
103  {
104  /** The largest positive value size_t can hold. */
105  static constexpr size_t npos = static_cast<size_t>(-1);
106 
107  /** The length of the string excluding the terminating '\0' character. */
108  static constexpr size_t size = N;
109 
110  /**
111  * The statically-sized character buffer used for storing the string.
112  */
113  char data[N + 1];
114 
115  /**
116  * Creates an empty const_string.
117  */
118  constexpr const_string() noexcept
119  : data{}
120  {
121  }
122 
123  /**
124  * Creates a copy of a const_string.
125  */
126  constexpr const_string(const const_string<N>& other) noexcept
127  : data{}
128  {
129  for (size_t i = 0; i < N; i++)
130  data[i] = other.data[i];
131  }
132 
133  /**
134  * Creates a const_string by copying the contents of data.
135  */
136  constexpr const_string(const char(&data)[N + 1]) noexcept
137  : data{}
138  {
139  for (size_t i = 0; i < N; i++)
140  this->data[i] = data[i];
141  }
142 
143  /**
144  * Explicitly converts to const char*.
145  */
146  explicit constexpr operator const char*() const noexcept
147  {
148  return data;
149  }
150 
151  /**
152  * Explicitly converts to std::string.
153  */
154  explicit operator std::string() const noexcept
155  {
156  return data;
157  }
158 
159  /**
160  * Returns a pointer to the contained zero-terminated string.
161  */
162  constexpr const char* c_str() const noexcept
163  {
164  return data;
165  }
166 
167  /**
168  * Returns the contained string as an std::string.
169  */
170  std::string str() const noexcept
171  {
172  return data;
173  }
174 
175  /**
176  * A constexpr version of std::string::substr.
177  */
178  template <size_t Pos, size_t Count = npos>
179  constexpr auto substr() const noexcept
180  {
181  static_assert(Pos <= N);
182  constexpr size_t NewSize = std::min(Count, N - Pos);
183 
184  char buf[NewSize + 1]{};
185  for (size_t i = 0; i < NewSize; i++) {
186  buf[i] = data[Pos + i];
187  }
188 
189  return const_string<NewSize>(buf);
190  }
191 
192  };
193 
194  /**
195  * Creates an empty instance of const_string<N>
196  */
197  constexpr const_string<0> make_const_string() noexcept
198  {
199  return {};
200  }

202  /**
203  * Creates an instance of const_string<N>
204  */
205  template <size_t N>
206  constexpr const_string<N - 1> make_const_string(const char(&str)[N]) noexcept
207  {
208  return str;
209  }

211  /**
212  * Concatenates two const_strings together.
213  */
214  template <size_t N, size_t M>
215  constexpr const_string<N + M> operator+(const const_string<N>& a, const const_string<M>& b) noexcept
216  {
217  char data[N + M + 1] { };
218  for (size_t i = 0; i < N; i++)
219  data[i] = a.data[i];
220  for (size_t i = 0; i < M; i++)
221  data[N + i] = b.data[i];
222  return data;
223  }
224 
225  /**
226  * Concatenates a const_string with a C-style string.
227  */
228  template <size_t N, size_t M>
229  constexpr const_string<N + M - 1> operator+(const const_string<N>& a, const char(&b)[M]) noexcept
230  {
231  return a + make_const_string(b);
232  }
233 
234  /**
235  * Concatenates a C-style string with a const_string.
236  */
237  template <size_t N, size_t M>
238  constexpr const_string<N + M - 1> operator+(const char(&a)[N], const const_string<M>& b) noexcept
239  {
240  return make_const_string(a) + b;
241  }
242 
243  /**
244  * Compares two const_strings for equality.
245  */
246  template <size_t N, size_t M>
247  constexpr bool operator==(const const_string<N>& a, const const_string<M>& b) noexcept
248  {
249  if constexpr (N != M) {
250  return false;
251  }
252  else {
253  for (size_t i = 0; i < M; i++) {
254  if (a.data[i] != b.data[i]) {
255  return false;
256  }
257  }
258  return true;
259  }
260  }

264  */
265  template <size_t N, size_t M>
266  constexpr bool operator!=(const const_string<N>& a, const const_string<M>& b) noexcept
267  {
