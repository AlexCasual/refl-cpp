861  template <typename F, typename... Carry>
862  constexpr auto eval_in_order_to_tuple(type_list<>, std::index_sequence<>, F&&, Carry&&... carry)
863  {
864  if constexpr (sizeof...(Carry) == 0) return std::tuple<>{};
865  else return std::make_tuple(std::forward<Carry>(carry)...);
866  }
867 
868  // This whole jazzy workaround is needed since C++ does not specify
869  // the order in which function arguments are evaluated and this leads
870  // to incorrect order of evaluation (noticeable when using indexes).
871  // Otherwise we could simply do std::make_tuple(f(Ts{}, Idx)...).
872  template <typename F, typename T, typename... Ts, size_t I, size_t... Idx, typename... Carry>
873  constexpr auto eval_in_order_to_tuple(type_list<T, Ts...>, std::index_sequence<I, Idx...>, F&& f, Carry&&... carry)
874  {
875  static_assert(std::is_trivial_v<T>, "Cannot synthesize a value of type T while iterating type_list<T, ...>!");
876 
877  if constexpr (std::is_invocable_v<F, T, size_t>) {
878  return eval_in_order_to_tuple(
879  type_list<Ts...>{},
880  std::index_sequence<Idx...>{},
881  std::forward<F>(f),
882  std::forward<Carry>(carry)..., // carry the previous results over
883  f(T{}, I) // pass the current result after them
884  );
885  }
886  else {
887  return eval_in_order_to_tuple(
888  type_list<Ts...>{},
889  std::index_sequence<Idx...>{},
890  std::forward<F>(f),
891  std::forward<Carry>(carry)..., // carry the previous results over
892  f(T{}) // pass the current result after them
893  );
894  }
895  }
896  }
