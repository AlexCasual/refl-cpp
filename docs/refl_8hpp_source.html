1062  else return std::make_tuple(std::forward<Carry>(carry)...);

1065  // This whole jazzy workaround is needed since C++ does not specify
1066  // the order in which function arguments are evaluated and this leads
1067  // to incorrect order of evaluation (noticeable when using indexes).
1068  // Otherwise we could simply do std::make_tuple(f(Ts{}, Idx)...).
1069  template <typename F, typename T, typename... Ts, size_t I, size_t... Idx, typename... Carry>
1070  constexpr auto eval_in_order_to_tuple(type_list<T, Ts...>, std::index_sequence<I, Idx...>, F&& f, Carry&&... carry)
1071  {
1072  static_assert(std::is_trivial_v<T>, "Cannot synthesize a value of type T while iterating type_list<T, ...>!");
1073 
1074  if constexpr (std::is_invocable_v<F, T, size_t>) {
1075  return eval_in_order_to_tuple(
1076  type_list<Ts...>{},
1077  std::index_sequence<Idx...>{},
1078  std::forward<F>(f),
1079  std::forward<Carry>(carry)..., // carry the previous results over
1080  f(T{}, I) // pass the current result after them
1081  );
1082  }
1083  else {
1084  return eval_in_order_to_tuple(
1085  type_list<Ts...>{},
1086  std::index_sequence<Idx...>{},
1087  std::forward<F>(f),
1088  std::forward<Carry>(carry)..., // carry the previous results over
1089  f(T{}) // pass the current result after them
1090  );
1091  }
1092  }
