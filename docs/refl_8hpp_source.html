827  // the order in which function arguments are evaluated and this leads
828  // to incorrect order of evaluation (noticeable when using indexes).

835  type_list<Ts...>{},
836  std::index_sequence<Idx...>{},
837  std::forward<F>(f),
838  std::forward<Carry>(carry)..., // carry the previous results over
839  f(T{}, I) // pass the current result after them
840  );

844  type_list<Ts...>{},
845  std::index_sequence<Idx...>{},
846  std::forward<F>(f),
847  std::forward<Carry>(carry)..., // carry the previous results over
848  f(T{}) // pass the current result after them
