726  // This whole jazzy workaround is needed since C++ does not specify
727  // the order in which function arguments are evaluated and this leads
728  // to incorrect order of evaluation (noticeable when using indexes).
729  // Otherwise we could simply do std::make_tuple(f(Ts{}, Idx)...).
730  template <typename F, typename T, typename... Ts, size_t I, size_t... Idx, typename... Carry>
731  constexpr auto eval_in_order_to_tuple(type_list<T, Ts...>, std::index_sequence<I, Idx...>, F&& f, Carry&&... carry)
732  {
733  if constexpr (std::is_invocable_v<F, T, size_t>) {
734  return eval_in_order_to_tuple(
735  type_list<Ts...>{},
736  std::index_sequence<Idx...>{},
737  std::forward<F>(f),
738  std::forward<Carry>(carry)..., // carry the previous results over
739  f(T{}, I) // pass the current result after them
740  );
741  }
742  else {
743  return eval_in_order_to_tuple(
744  type_list<Ts...>{},
745  std::index_sequence<Idx...>{},
746  std::forward<F>(f),
747  std::forward<Carry>(carry)..., // carry the previous results over
748  f(T{}) // pass the current result after them
749  );
750  }
751  }
